/**
 * Created by admin on 2018/4/5.
 */
const http = require('https');
const cheerio = require('cheerio');
const DB = require("./mysql");
const Common = require("./lib/common");

const baseUrl = 'https://www.jianshu.com/';

const Spider = {
    /**
     * 获取首页文章的地址
     */
    getIndexUrls: () => {
        console.log("开始抓取：----------");
        http.get(baseUrl, function(res) {
            let html = "",
                $, urls = [];

            res.on('data', function(data) {
                html += data;
            });
            res.on('end', function() {
                // 要执行的操作
                let $listCon;
                $ = cheerio.load(html);
                $listCon = $("#list-container");
                $listCon.find("li").each(function() {
                    urls.push($(this).find(".title").attr("href"));
                });
                urls.forEach((url) => {
                    Spider.getArticle(baseUrl + url, url);
                });
            })
        }).on('error', function() {
            console.warn("获取数据失败！");
        })
    },

    /**
     * 获取文章信息
     * @param url  文章地址
     * @param articleId  文章编号
     */
    getArticle: (url, articleId) => {
        console.log("抓取文章：----------");
        http.get(url, function(res) {
            let html = "",
                $;

            // 获取页面数据
            res.on('data', function(data) {
                html += data;
            });
            // 数据获取结束
            res.on('end', function() {
                // 要执行的操作
                $ = cheerio.load(html, {
                    decodeEntities: false
                });
                const $article = $(".article"),
                    $thumbnail = $(".image-view").find("img"),
                    $notebook = $(".notebook"),
                    aid = articleId.replace("/p/", ""),
                    subjectUrl = $notebook.attr("href"),
                    sid = subjectUrl.replace("/nb/", ""),
                    accountUrl = $article.find('.name').find("a").attr("href"),
                    account = accountUrl.replace("/u/", ""),
                    title = $article.find(".title").text(),
                    content = $article.find(".show-content-free").html(),
                    date = $article.find('.publish-time').text().replace(/\./g, "/").slice(0, 16),
                    countLike = Math.round((Math.random() + 1) * Math.pow(10, Math.round(Math.random() * 5))),
                    countComment = Math.round((Math.random() + 1) * Math.pow(10, Math.round(Math.random() * 3))),
                    countRead = Math.round((Math.random() + 1) * Math.max(countLike, countComment)),
                    thumbnail = "https:" + $thumbnail.attr("data-original-src");
                 DB.addArticle({
                     aid: aid,
                     sid: sid,
                     account: account,
                     title: title,
                     content: content,
                     date: date,
                     like: countLike,
                     comment: countComment,
                     read: countRead,
                     picture: thumbnail
                 })
                Spider.getSubject(subjectUrl);
                Spider.getAuthor(account);
            })
        }).on('error', function() {
            console.warn("获取数据失败！");
        })
    },

    /**
     * 获取专题信息
     * @param subject 专题号
     */
    getSubject: (subject) => {
        console.log("抓取专题信息：----------");
        http.get(baseUrl + subject, (res) => {
            let html = "",
                $;

            res.on("data", (data) => {
                html += data;
            });
            res.on("end", () => {
                $ = cheerio.load(html, {
                    decodeEntities: false
                });
                const $main = $(".main"),
                    $aside = $(".aside"),
                    sid = subject.replace("/nb/", ""),
                    name = $main.find(".name").text(),
                    brief = $main.find(".summary").text(),
                    thumbnail = $main.find(".avatar-collection").find("img").attr("src"),
                    belongUrl = $aside.find(".name").attr("href"),
                    belong = belongUrl.replace("/u/", ""),
                    info = $main.find(".info").text(),
                    infos = info.split(" · "),
                    date = new Date();
                let countFocus = 0;

                countFocus = infos[2];
                countFocus = countFocus.replace("人关注", "");
                DB.addSubject({
                    sid: sid,
                    name: name,
                    brief: brief,
                    thumbnail: thumbnail,
                    belong: belong,
                    countFocus: countFocus,
                    date: date
                })
            })
        })
    },

    /**
     * 获取作者信息页内容：作者信息以及作者的文章
     * @param account 作者账号
     */
    getAuthor: (account) => {
        console.log("抓取作者信息：----------");
        http.get(baseUrl + "u/" + account, function(res) {
            let html = "",
                $;

            res.on("data", function(data) {
                html += data;
            });
            res.on("end", function() {
                $ = cheerio.load(html, {
                    decodeEntities: false
                });
                let $user = $(".main-top"),
                    $meta = $(".meta-block").find("p"),
                    $list = $("#list-container"),
                    password = new Buffer(account),
                    nickname = $user.find(".title").text(),
                    brief = $(".js-intro").text(),
                    countSubject = $(".aside").find(".title").length,
                    countFocus = 0,
                    countFans = 0,
                    countArticle = 0,
                    countWord = 0,
                    countLiked = 0,
                    urls = [];

                $meta.each(function(index) {
                    switch(index) {
                        case 0: {countFocus = $(this).text();break;}
                        case 1: {countFans = $(this).text();break;}
                        case 2: {countArticle = $(this).text();break;}
                        case 3: {countWord = $(this).text();break;}
                        case 4: {countLiked = $(this).text();break;}
                    }
                });
                $list.find(".title").each(function() {
                    urls.push($(this).attr("href"));
                });
                DB.addAuthor({
                     account: account.slice(0, 12),
                     password: password.reverse().toString("base64"),
                     nickname: nickname,
                     brief: brief,
                     countFocus: countFocus,
                     countFans: countFans,
                     countArticle: countArticle,
                     countWord: countWord,
                     countLiked: countLiked,
                     countSubject: countSubject,
                     countLike: 1
                });
                urls.forEach((url) => {
                     Spider.getAuthorContent(baseUrl + url);
                     Spider.getAuthorAction(account);
                });
            })
        })
    },

    /**
     * 获取作者文章内容
     * @param url 作者文章地址
     */
    getAuthorContent: (url) => {
        http.get(url, function(res) {
            let html = "",
                $;

            // 获取页面数据
            res.on('data', function(data) {
                html += data;
            });
            // 数据获取结束
            res.on('end', function() {
                // 要执行的操作
                $ = cheerio.load(html, {
                    decodeEntities: false
                });
                let $article = $(".article"),
                    title = $article.find(".title").text(),
                    content = $article.find(".show-content-free").html(),
                    accountDom = $article.find('.name').find("a").attr("href"),
                    name = $article.find(".name").text(),
                    date = $article.find('.publish-time').text(),
                    account = "",
                    countLike = Math.round((Math.random() + 1) * Math.pow(10, Math.round(Math.random() * 5))),
                    countComment = Math.round((Math.random() + 1) * Math.pow(10, Math.round(Math.random() * 3))),
                    countRead = Math.round((Math.random() + 1) * Math.max(countLike, countComment));

                account = accountDom.split("/");
            })
        }).on('error', function() {
            console.log("获取数据失败！");
        })
    },

    /**
     * 获取作者动态
     * @param account 作者账号
     */
    getAuthorAction: (account) => {
        http.get(baseUrl + "users/" + account + "/timeline", (res) => {
            let html = "",
                $;

            res.on("data", (data) => {
                html += data;
            });
            res.on("end", () => {
                $ = cheerio.load(html, {
                    decodeEntities: false
                });
                const $container = $("#list-container"),
                    $item = $container.find("li");

                $item.each(function() {
                    const $this = $(this),
                        $title = $this.find(".title"),
                        $info = $this.find(".info").children("span"),
                        keyStr = $title.attr("href"),
                        value = $title.text(),
                        typeStr = $info.attr("data-type"),
                        $date = $info.attr("data-datetime"),
                        date = $date.replace("T", " ").slice(0, 16);
                    let key = "",
                        type = 0;

                    switch(typeStr) {
                        case "like_user": {
                            key = keyStr.replace("/u/", "");
                            type = 0;
                            break;
                        }
                        case "like_note": {
                            key = keyStr.replace("/p/", "");
                            type = 1;
                            break;
                        }
                        case "share_note": {
                            key = keyStr.replace("/p/", "");
                            type = 2;
                            break;
                        }
                        case "like_notebook": {
                            key = keyStr.replace("/nb/", "");
                            type = 3;
                            break;
                        }
                        case "like_collection": {
                            key = keyStr.replace("/c/", "");
                            type = 3;
                            break;
                        }
                        case "comment_note": {
                            type = 4;
                            break;
                        }
                    }
                    DB.addAction({
                        id: account + key,
                        key: key,
                        value: value,
                        account: account,
                        type: type,
                        date: date
                    })
                })
            })
        })
    }
};

function main() {
    Spider.getIndexUrls();
}

main(); // 运行主函数
